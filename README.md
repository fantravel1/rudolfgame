### Shake Detection
```javascript
let lastShake = 0;
const SHAKE_THRESHOLD = 15;

window.addEventListener('devicemotion', (e) => {
  const acc = e.accelerationIncludingGravity;
  const total = Math.abs(acc.x) + Math.abs(acc.y) + Math.abs(acc.z);
  
  if (total > SHAKE_THRESHOLD && Date.now() - lastShake > 2000) {
    lastShake = Date.now();
    activateNoseBoost();
    vibratePhone([100, 50, 100]); // Boost feedback
  }
});
```
